This post aims to give some basic facts on coronavirus epidemic which took place in the last couple of months and spreaded around all the world. The main objective is to focus on the economic consequences.

knitr::opts_chunk$set(message = FALSE, warning=FALSE, echo = TRUE, cache = FALSE)
library(tidyverse)
library(lubridate)
theme_set(theme_light())
graphics.off()
dat <- tibble(Date = today() + days(0:99),
              Vals = rnorm(100))

ggplot(dat, aes(Date, Vals))+
  geom_line(col = "black")+
  geom_hline(yintercept = 0, col = "red", linetype = "dashed")+
  geom_hline(yintercept = 1, col = "blue", linetype = "dashed")+
  geom_hline(yintercept = -1, col = "black", linetype = "dashed")
